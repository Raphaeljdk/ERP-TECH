Método INSERT genérico
 public T save(T entidade) {
 String sql = "INSERT INTO " + getTableName() + " VALUES (?)"; Simplificado
 try (PreparedStatement stmt = getConnection().prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {
 setParametersForInsert(stmt, entity);
 stmt.executeUpdate();
 
ResultSet rs = stmt.getGeneratedKeys();
 se (rs.next()) {
 setId(entity, rs.getInt(1));
 }
 retornar a entidade;
 } catch (SQLException e) {
 lançar um novo RuntimeException ("Erro ao salvar!", e);
 }
 }
 
abstrato protegido void setId(T entity, ID ID);

try (Statement stmt = getConnection().createStatement();

classe pública ValidadorCPF {
 
validar booleano estático público(String cpf) {
 Remove caracteres não numéricos
 cpf = cpf.replaceAll("[^0-9]", "");
 
Verifica se tem 11 dígitos
 se (cpf.length() != 11) retorna false;
 
Verifica se todos são iguais (111.111.111-11)
 se (cpf.matches("(\\d)\\1{10}")) retorna falso;
 
Validação dos dígitos
 try {
 int[] números = novo int[11];
 para (int i = 0; i < 11; i++) {
 numeros[i] = Integer.parseInt(cpf.substring(i, i + 1));
 }
 
Primeiro dígito verificador
 int soma = 0;
 para (int i = 0; i < 9; i++) {
 soma += números[i] * (10 - i);
 }
 int dígito1 = 11 - (soma % 11);
 se (dígito1 > 9) dígito1 = 0;
 
Segundo dígito verificador
 Soma = 0;
 para (int i = 0; i < 10; i++) {
 soma += números[i] * (11 - i);
 }
 int dígito2 = 11 - (soma % 11);
 se (dígito2 > 9) dígito2 = 0;
 
números de retorno[9] == digito1 & números[10] == digito2;
 
} catch (NumberFormatException e) {
 retorne falso;
 }
 }
 
public static String formatar(String cpf) {
 cpf = cpf.replaceAll("[^0-9]", "");
 se (cpf.length() != 11) retorna cpf;
 
return String.format("%s.%s.%s-%s",
 cpf.substring(0, 3),
 cpf.substring(3, 6),
 cpf.substring(6, 9),
 cpf.substring(9, 11)
 );
 }
}
